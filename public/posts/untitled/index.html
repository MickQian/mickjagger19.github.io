//source https://iquilezles.org/articles/distfunctions
float boxSDF( vec3 point, vec3 bounds )
{
    vec3 q = abs(point) - bounds;
    return length(max(q, 0.0)) + min(max(q.x, max(q.y, q.z)), 0.0);
}

float boxSDF2D( in vec2 p, in vec2 b )
{
    vec2 d = abs(p)-b;
    return length(max(d,0.0)) + min(max(d.x,d.y),0.0);
}

float orientedBoxSDF2D( in vec2 p, in vec2 a, in vec2 b, float th )
{
    float l = length(b-a);
    vec2  d = (b-a)/l;
    vec2  q = (p-(a+b)*0.5);
          q = mat2(d.x,-d.y,d.y,d.x)*q;
          q = abs(q)-vec2(l,th)*0.5;
    return length(max(q,0.0)) + min(max(q.x,q.y),0.0);    
}

// Generate color for strips
vec3 generateColor(float t) {
    vec3 col = vec3(0);
    
    col.r = noise11(t);
    col.g = noise11(t + 2.2321);
    
    float rg = col.r + col.g;
    
    
    col.b = noise11(t + 8.32132);
    
    
    if (col.r + col.g + col.b <= 0.3) {
    
         col.b += noise21(vec2(col.r, col.g));
    }
    
   // col = vec3(1.0);
    
    return col;

}

vec4 map(vec2 point)

    float minSDF = 1000000.0;

    for (float x = -2.0; x <= 2.1; x+=0.3)
    {
        for (float y = -0.9; y <= 0.9; y+=0.9)
        {
        
				vec2 tempRootPoint = rootPoint + vec2(x * LINE_SPACE_X,  y * LINE_SPACE_Y);
                
                tempRootPoint.y = 0.;
                
           //     tempRootPoint = vec2(0.0, 0.0);
                
                vec2 lineHash = hash22(tempRootPoint);
                
                float hash = hash21(tempRootPoint.xy) - 0.5;

                boxSize.y = max(0.05, 1.42 - boxSize.x );

/// Calculate the color and minimum distane at given point
vec4 objectSDF(vec3 point)
{
   // point += spaceBounding(point);
    
    vec4 lines = map(point.xy);
    
    
    /// Calculate the closest distance
    float cylinder = cylinderSDF(point, BOUNDING_CYLINDER);
    
    // distance to the outside bounding cylinder
    float insideCylinder = -cylinderSDF(point, INSIDE_CYLINDER);
    
    
    float object = multiplyObjects(lines.a, cylinder);
    object = lines.a;
   // object = multiplyObjects(object, insideCylinder);
    
    
 	return vec4(lines.rgb, object);
}
